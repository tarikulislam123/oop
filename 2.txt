#include <iostream>
using namespace std;
class car
{
    string name;
    int speed;
    int gear;
public:
    car(string name, int speed, int gear)
    {
        this->name = name;
        this->speed = speed;
        this->gear = gear;
    }

    car(string name, int speed); //declaration

    void getValue()
    {
        cout<<"Car Name: "<<name<<endl;
        cout<<"Speed: "<<speed<<endl;
        cout<<"Gear: "<<gear<<endl;
    }

 void updateGear(int x); //declaration

};


void car :: updateGear(int x) //definition
{
    this->gear = gear + x;
}

 car::car(string name, int speed) //definition
{
    this->name = name;
    this->speed = speed;
    this->gear = 0;
}



int main()
{
    car *bmw = new car("BMW", 20, 15);
